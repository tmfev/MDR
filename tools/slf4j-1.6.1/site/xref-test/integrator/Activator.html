22  package integrator;
23  
24  import org.osgi.framework.Bundle;
25  import org.osgi.framework.BundleActivator;
26  import org.osgi.framework.BundleContext;
27  import org.slf4j.Logger;
28  import org.slf4j.LoggerFactory;
29  
30  /**
31   * A BundleActivator which invokes slf4j loggers
32   * @author Ceki G&uuml;lc&uuml;
33   *
34   */
35  public class Activator implements BundleActivator {
36  
37    private BundleContext m_context = null;
38  
39    public void start(BundleContext context) {
40      Logger logger = LoggerFactory.getLogger(this.getClass());
41      logger.info("Activator.start()");
42      m_context = context;
43    }
44  
45    public void stop(BundleContext context) {
46      m_context = null;
47      Logger logger = LoggerFactory.getLogger(this.getClass());
48      logger.info("Activator.stop");
49    }
50  
51    public Bundle[] getBundles() {
52      if (m_context != null) {
53        return m_context.getBundles();
54      }
55      return null;
56    }
57  }
