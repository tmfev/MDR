19  public class XLoggerFactory {

25     * @return

27    public static XLogger getXLogger(String name) {
28      return new XLogger(LoggerFactory.getLogger(name));
29    }

35     * @param clazz
36     * @return

38    @SuppressWarnings("unchecked")
