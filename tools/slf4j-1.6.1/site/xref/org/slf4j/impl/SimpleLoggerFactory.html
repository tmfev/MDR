34  package org.slf4j.impl;
35  
36  import java.util.HashMap;
37  import java.util.Map;
38  
39  import org.slf4j.Logger;
40  import org.slf4j.ILoggerFactory;
41  
42  /**
43   * An implementation of {@link ILoggerFactory} which always returns
44   * {@link SimpleLogger} instances.
45   * 
46   * @author Ceki G&uuml;lc&uuml;
47   */
48  public class SimpleLoggerFactory implements ILoggerFactory {
49  
50    final static SimpleLoggerFactory INSTANCE = new SimpleLoggerFactory();
51  
52    Map loggerMap;
53  
54    public SimpleLoggerFactory() {
55      loggerMap = new HashMap();
56    }
57  
58    /**
59     * Return an appropriate {@link SimpleLogger} instance by name.
60     */
61    public Logger getLogger(String name) {
62      Logger slogger = null;
63      // protect against concurrent access of the loggerMap
64      synchronized (this) {
65        slogger = (Logger) loggerMap.get(name);
66        if (slogger == null) {
67          slogger = new SimpleLogger(name);
68          loggerMap.put(name, slogger);
69        }
70      }
71      return slogger;
72    }
73  }
